// Phần Script trong file login.html document.getElementById("loginForm").onsubmit = async (e) => { e.preventDefault(); const btn = document.getElementById("loginBtn"); const msgDiv = document.getElementById("message"); const student_id = document.getElementById("student_id").value.trim(); const password = document.getElementById("password").value; btn.disabled = true; btn.innerText = "Đang xác thực..."; try { const res = await fetch('/api/login', { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ student_id, password }) }); const data = await res.json(); if (res.ok) { localStorage.setItem("student_token", data.token); localStorage.setItem("student_id", data.student_id); localStorage.setItem("student_name", data.name); window.location.href = "index.html"; } else { msgDiv.style.color = "red"; msgDiv.innerText = "❌ " + data.error; btn.disabled = false; btn.innerText = "Đăng nhập"; } } catch (err) { msgDiv.innerText = "⚠️ Lỗi kết nối Server."; btn.disabled = false; } };
